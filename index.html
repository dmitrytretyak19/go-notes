ПЕРЕМЕННЫЕ var и :=

Переменная — коробка с именем. Внутри хранится значение.

"25" — текст, математика не работает!
25 — число. Разница важна!

🔄 УР. 4 Циклы 2 карточки ▾
ЦИКЛ for — три части

В Go один вид цикла — for. Повторяет код нужное количество раз.

// Обычный цикл от 1 до 5
for i := 1; i <= 5; i++ {
    fmt.Println(i)
}

// Шаг по 5: 5, 10, 15, 20, 25
for i := 5; i <= 25; i += 5 {
    fmt.Println(i)
}
i := 1 — откуда начать
i <= 5 — когда остановиться
i++ — прибавить 1 каждый шаг
УПРАВЛЕНИЕ break / continue / % 
// break — остановить цикл полностью
for i := 1; i <= 10; i++ {
    if i == 5 { break }
    fmt.Println(i) // 1 2 3 4
}

// continue — пропустить шаг
for i := 1; i <= 5; i++ {
    if i == 3 { continue }
    fmt.Println(i) // 1 2 4 5
}

// % остаток — только чётные числа
for i := 1; i <= 10; i++ {
    if i%2 != 0 { continue }
    fmt.Println(i) // 2 4 6 8 10
}
break срабатывает ДО вывода! i==5 с break → выведет 1 2 3 4

ФУНКЦИИ Создание и вызов

Пишешь один раз — вызываешь сколько угодно.

// Без возврата
func приветствие(имя string) {
    fmt.Println("Привет,", имя)
}

// Два параметра
func инфо(имя string, возраст int) {
    fmt.Println(имя, "возраст:", возраст)
}

// Вызов
приветствие("Дмитрий")
инфо("Алибек", 30)
RETURN Возврат значения 
// Возвращает число
func сложить(а int, б int) int {
    return а + б
}

// Возвращает bool
func isAdult(возраст int) bool {
    if возраст >= 18 {
        return true
    }
    return false
}

// Использование
результат := сложить(5, 3)
fmt.Println(результат) // 8
return числа → пиши int
return текста → пиши string
return true/false → пиши bool
нет return → нет типа

STRUCT Создание и использование

Коробка с несколькими полями разных типов. Как анкета.

// Тип структуры — с Большой буквы!
type User struct {
    Name string
    Age  int
    City string
}

// Переменная — с маленькой
user := User{
    Name: "Дмитрий",
    Age:  25,
    City: "Гродно",
}

// Получить поле — через точку
fmt.Println(user.Name)

// Изменить поле
user.Age = 26
Тип/поля → с Большой (User, Name)
Переменные → с маленькой (user)
Доступ → через точку user.Name

🔌 УР. 9 Интерфейсы 2 карточки ▾
INTERFACE Контракт

Позволяет одной функции работать с разными типами.

// Интерфейс — контракт
type Animal interface {
    Sound() string
}

// Структуры
type Dog struct { Name string }
type Cat struct { Name string }

// Методы
func (d Dog) Sound() string { return "Гав!" }
func (c Cat) Sound() string { return "Мяу!" }

// Одна функция для всех Animal
func makeSound(a Animal) {
    fmt.Println(a.Sound())
}

makeSound(Dog{Name: "Рекс"})
makeSound(Cat{Name: "Мурзик"})
Dog умеет Sound → Dog является Animal ✅
Cat умеет Sound → Cat является Animal ✅
МЕТОД Функция с владельцем 
// Обычная функция
func Sound() string { ... }

// Метод — привязан к структуре
func (d Dog) Sound() string {
    return "Гав! Я " + d.Name
}

// d — короткое имя для Dog
// По традиции — первая буква типа:
// Dog → d, Cat → c, User → u
(d Dog) → функция принадлежит Dog
d.Name → обращаемся к полю через d
❌ УР. 10 Ошибки 1 карточка ▾
ERROR Обработка ошибок

Функции возвращают два значения — результат и ошибку. Всегда проверяй!

import (
    "errors"
    "fmt"
)

// Функция с двумя возвратами
func разделить(а int, б int) (int, error) {
    if б == 0 {
        return 0, errors.New("нельзя на ноль!")
    }
    return а / б, nil
}

// Всегда проверяй ошибку!
результат, ошибка := разделить(10, 2)
if ошибка != nil {
    fmt.Println("Ошибка:", ошибка)
    return
}
fmt.Println(результат) // 5
nil — "ничего", ошибки нет ✅
ошибка != nil — ошибка есть ❌
errors.New("текст") — создать ошибку

GIT Основные команды

Git — машина времени для кода. Сохраняет историю всех изменений.

# Проверить версию Git
git --version

# Настроить имя и email (один раз)
git config --global user.name "Dmitry"
git config --global user.email "email@gmail.com"

# Создать репозиторий в папке
git init

# Посмотреть что изменилось
git status

# Добавить все файлы
git add .

# Сохранить с описанием
git commit -m "описание что изменил"

# Посмотреть историю коммитов
git log --oneline

# Вернуться к старой версии
git checkout ID_коммита

# Вернуться к последней версии
git checkout main
GIT Как работает коммит

Три шага которые повторяешь каждый раз когда делаешь изменения.

КомандаЧто делает

git statusЧто изменилось

DOCKER Что такое Docker

Docker упаковывает твою программу вместе со всем что ей нужно — Go, библиотеки, настройки — в один контейнер. Работает одинаково везде.

ПонятиеЧто этоАналогия
Image (образ)шаблон контейнерарецепт блюда
Containerзапущенный образготовое блюдо
Dockerfileинструкция сборкисписок ингредиентов
Docker Hubбиблиотека образовмагазин рецептов
Порядок работы:
1. Написал Dockerfile (рецепт)
2. docker build (приготовил)
3. docker run (запустил)
DOCKERFILE Файл инструкций

Dockerfile — файл без расширения в корне проекта. Содержит инструкции как собрать образ.

# Базовый образ — на основе чего строим
FROM golang:1.22
 
# Рабочая папка внутри контейнера
WORKDIR /app
 
# Копируем весь код внутрь контейнера
COPY . .
 
# Команда которая запустится при старте
CMD ["go", "run", "main.go"]
FROM — на основе какого образа строим
WORKDIR — рабочая папка в контейнере
COPY . . — копируем код внутрь
CMD — команда запуска
КОМАНДЫ Основные команды Docker 
# Проверить версию Docker
docker --version
 
# Запустить тестовый контейнер
docker run hello-world
 
# Собрать образ из Dockerfile
# -t = tag (название образа)
# . = использовать текущую папку
docker build -t my-go-app .
 
# Запустить контейнер
docker run my-go-app
 
# Список всех образов
docker images
 
# Список запущенных контейнеров
docker ps
 
# Список всех контейнеров
docker ps -a
 
# Остановить контейнер
docker stop ID_контейнера
 
# Удалить контейнер
docker rm ID_контейнера
 
# Удалить образ
docker rmi my-go-app
-t название даёт образу понятное имя вместо длинного ID типа sha256:bba8fd9c...
