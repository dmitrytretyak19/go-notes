🔄 УР. 4 Циклы
▾
for — три части
В Go один вид цикла — for. Повторяет код нужное количество раз.
// Обычный цикл от 1 до 5
for i := 1; i <= 5; i++ {
    fmt.Println(i)
}

// Шаг по 5: 5, 10, 15, 20, 25
for i := 5; i <= 25; i += 5 {
    fmt.Println(i)
}
i := 1 — откуда начать
i <= 5 — когда остановиться
i++ — прибавить 1 каждый шаг
break / continue / %
// break — остановить цикл полностью
for i := 1; i <= 10; i++ {
    if i == 5 { break }
    fmt.Println(i) // 1 2 3 4
}

// continue — пропустить шаг
for i := 1; i <= 5; i++ {
    if i == 3 { continue }
    fmt.Println(i) // 1 2 4 5
}

// % остаток — только чётные числа
for i := 1; i <= 10; i++ {
    if i%2 != 0 { continue }
    fmt.Println(i) // 2 4 6 8 10
}
break срабатывает ДО вывода! i==5 с break → выведет 1 2 3 4

Создание и вызов
Пишешь один раз — вызываешь сколько угодно.
// Без возврата
func приветствие(имя string) {
    fmt.Println("Привет,", имя)
}

// Два параметра
func инфо(имя string, возраст int) {
    fmt.Println(имя, "возраст:", возраст)
}

// Вызов
приветствие("Дмитрий")
инфо("Алибек", 30)
Типы параметров указываются после их имени (имя string).
Возврат значения
// Возвращает число
func сложить(а int, б int) int {
    return а + б
}

// Возвращает bool
func isAdult(возраст int) bool {
    if возраст >= 18 {
        return true
    }
    return false
}

// Использование
результат := сложить(5, 3)
fmt.Println(результат) // 8
return числа → пиши int
return текста → пиши string
return true/false → пиши bool
нет return → нет типа

🔌 УР. 9 Интерфейсы
▾
Контракт
Позволяет одной функции работать с разными типами.
// Интерфейс — контракт
type Animal interface {
    Sound() string
}

// Структуры
type Dog struct { Name string }
type Cat struct { Name string }

// Методы
func (d Dog) Sound() string { return "Гав!" }
func (c Cat) Sound() string { return "Мяу!" }

// Одна функция для всех Animal
func makeSound(a Animal) {
    fmt.Println(a.Sound())
}

makeSound(Dog{Name: "Рекс"})
makeSound(Cat{Name: "Мурзик"})
Dog умеет Sound → Dog является Animal ✅
Cat умеет Sound → Cat является Animal ✅
Функция с владельцем
// Обычная функция
func Sound() string { ... }

// Метод — привязан к структуре
func (d Dog) Sound() string {
    return "Гав! Я " + d.Name
}

// d — короткое имя для Dog
// По традиции — первая буква типа:
// Dog → d, Cat → c, User → u
(d Dog) → функция принадлежит Dog
d.Name → обращаемся к полю через d
❌ УР. 10 Ошибки
▾
Обработка ошибок
Функции возвращают два значения — результат и ошибку. Всегда проверяй!
import (
    "errors"
    "fmt"
)

// Функция с двумя возвратами
func разделить(а int, б int) (int, error) {
    if б == 0 {
        return 0, errors.New("нельзя на ноль!")
    }
    return а / б, nil
}

// Всегда проверяй ошибку!
результат, ошибка := разделить(10, 2)
if ошибка != nil {
    fmt.Println("Ошибка:", ошибка)
    return
}
fmt.Println(результат) // 5
nil — "ничего", ошибки нет ✅
ошибка != nil — ошибка есть ❌
errors.New("текст") — создать ошибку

Docker упаковывает твою программу вместе со всем что ей нужно — Go, библиотеки, настройки — в один контейнер. Работает одинаково везде.

Запрос и ответ

          
          
        

        

          
HTTP — протокол общения браузера с сервером. Каждый раз когда открываешь сайт — браузер отправляет запрос, сервер отвечает.

          

            
1. Браузер отправляет запрос (Request) на сервер
2. Сервер обрабатывает запрос
3. Сервер отправляет ответ (Response)
4. Браузер показывает результат

            
          

          
URL = протокол + домен + путь
https://github.com/dmitrytretyak19
https — протокол, github.com — домен, /dmitrytretyak19 — путь

        

      

        

          
GET · POST · PUT · DELETE

          
          
        

        

          
Метод говорит серверу что именно ты хочешь сделать с данными.

МетодЧто делаетПример
GETПолучить данныеЗагрузить список пользователей
POSTСоздать новоеЗарегистрировать пользователя

Аналогия с рестораном:

      Первый HTTP сервер 
      

        
        
      

      ▾
    

        

          
Структура сервера

          
          
        

        

          
// handler — функция обработки запроса
// w — пишем ответ клиенту 
// r — запрос от клиента
// маршрут: при заходе на "/" вызвать handler
// запустить сервер на порту 8080

          

            
package main

import (
    "fmt"
    "net/http"
)

func handler(w http.ResponseWriter, r *http.Request) {
    fmt.Fprintln(w, "Привет от сервера!")
}

func main() {
    http.HandleFunc("/", handler)
    fmt.Println("Сервер запущен на порту 8080...")
   
  http.ListenAndServe(":8080", nil)
}
